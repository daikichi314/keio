41// 2つのベクトルがなす角を計算する関数
42double CalculateAngle(double x1, double y1, double z1, double x2, double y2, double z2) {
43 // ベクトルの内積
44 double dotProduct = x1 * x2 + y1 * y2 + z1 * z2;
45
46 // ベクトルの大きさ
47 double magnitude1 = std::sqrt(x1 * x1 + y1 * y1 + z1 * z1);
48 double magnitude2 = std::sqrt(x2 * x2 + y2 * y2 + z2 * z2);
49
50 // ベクトルの大きさが0の場合はエラー
51 if (magnitude1 == 0 || magnitude2 == 0) {
52 std::cerr << "Error: One or both vectors have zero magnitude." << std::endl;
53 return -1.0; // エラー値
54 }
55
56 // なす角（ラジアン）の計算
57 double cosalpha = dotProduct / (magnitude1 * magnitude2);
58 // 丸め誤差対策（cosThetaが[-1, 1]の範囲を超えることを防ぐ）
59 if (cosalpha > 1.0)
60 cosalpha = 1.0;
61 if (cosalpha < -1.0)
62 cosalpha = -1.0;
63
64 return cosalpha;
65}

67std::array<double, 3> ConvertToSpherical(
68 double x, double y, double z,
69 double x0, double y0, double z0 // 球座標の原点
70) {
71 double dx = x - x0, dy = y - y0, dz = z - z0;
72 double r = std::sqrt(dx * dx + dy * dy + dz * dz);
73 double theta = std::acos(-dy / r);
74 double phi = std::atan2(dz, dx);
75 return {r, theta, phi};
76}
77std::array<double, 3> ConvertToCartesian(
78 double r, double theta, double phi,
79 double x0, double y0, double z0 // 球座標の原点
80) {
81 double x = r * std::sin(theta) * std::cos(phi) + x0;
82 double y = -r * std::cos(theta) + y0;
83 double z = r * std::sin(theta) * std::sin(phi) + z0;
84 return {x, y, z};
85}

87void NormalizeAngles(double &theta, double &phi) {
88 if (theta < 0) {
89 theta = -theta;
90 phi += TMath::Pi();
91 }
92 if (theta > TMath::Pi()) {
93 theta = 2 * TMath::Pi() - theta;
94 phi += TMath::Pi();
95 }
96 // φ を [-π, π] に戻す
97 while (phi <= -TMath::Pi())
98 phi += 2 * TMath::Pi();
99 while (phi > TMath::Pi())
100 phi -= 2 * TMath::Pi();
101}

113std::vector<PMTData> findExpandedGroups(const std::vector<PMTData> &pmtData,
114 double tau,
115 int n,
116 double expand_before,
117 double expand_after) {
118
119 // ヒットを時刻順にpmtdataを並び替え（元の順序は保持されない）
120 std::vector<PMTData> sorted_hits = pmtData;
121 std::sort(sorted_hits.begin(), sorted_hits.end(),
122 [](const PMTData &a, const PMTData &b) { return a.t < b.t; });
123
124 std::set<int> used; // すでにグループに使われたヒットのインデックス
125 std::vector<PMTData> true_hits; // 実際に使うヒット
126
127 // unuseがあるかのチェック変数
128 int unuse = 0;
129 // ヒットごとにスキャン
130 for (size_t i = 0; i < sorted_hits.size(); ++i) {
131 // すでにグループに使われたらスキップ
132 if (used.count(i)) {
133 // std::cout << "this is used" << std::endl;
134 continue;
135 }
136
137 double time_i = sorted_hits[i].t;
138 // 現在のヒットを中心に、tau秒以内の他のヒットをカウント
139 int count = 0;
140 for (size_t j = i; j < sorted_hits.size(); ++j) {
141 // std::cout << " checking " << j << ", and this time is " << sorted_hits[j].t << std::endl;
142 if (sorted_hits[j].t - time_i > tau) {
143 // std::cout << "tau is over, tau is " << tau << std::endl;
144 continue; // tau秒を超えたら終了
145 }
146 count++;
147 // ヒット数がn未満ならスキップ（グループとみなさない）
148 }
149 if (count < n) {
150 count = 0;
151 continue;
152 }
153
154 // グループと認められたので、前後に拡張する
155 double t_start = time_i - expand_before;
156 double t_end = time_i + expand_after;
157 // 拡張した時間範囲にあるヒットをグループに追加
158 for (size_t j = 0; j < sorted_hits.size(); ++j) {
159 if (sorted_hits[j].t >= t_start && sorted_hits[j].t <= t_end) {
160 true_hits.push_back(sorted_hits[j]);
161 used.insert(j); // グループに使ったヒットを記録
162 }
163 }
164 break;
165 }
166 return true_hits;
167}
